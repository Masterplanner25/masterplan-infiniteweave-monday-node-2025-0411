import numpy as np
import pandas as pd
# Sample user interaction data
data = {
    "User": ["Alice", "Bob", "Charlie", "Alice", "Bob", "Charlie"],
    "Content": ["Post1", "Post2", "Post3", "Post3", "Post1", "Post2"],
    "Interaction": [5, 4, 3, 4, 5, 3],  # Higher score means higher interaction
}
# Create a DataFrame
df = pd.DataFrame(data)
# Step 1: Create a User-Content matrix
user_content_matrix = df.pivot_table(index="User", columns="Content", values="Interaction").fillna(0)
# Step 2: Calculate Content Similarity (Content-Based Filtering)
from sklearn.metrics.pairwise import cosine_similarity
content_similarity = cosine_similarity(user_content_matrix.T)  # Transpose for content similarity
content_similarity_df = pd.DataFrame(content_similarity, index=user_content_matrix.columns, columns=user_content_matrix.columns)
# Step 3: Predict content for a user (Collaborative + Content-Based)
def recommend_content(user, user_content_matrix, content_similarity_df):
    user_interactions = user_content_matrix.loc[user]
    unseen_content = user_interactions[user_interactions == 0].index  # Content not yet interacted with
    # Weighted recommendation scores
    scores = {}
    for content in unseen_content:
        similar_contents = content_similarity_df[content]
        scores[content] = np.dot(user_interactions, similar_contents)
    # Sort recommendations by score
    recommended = sorted(scores.items(), key=lambda x: x[1], reverse=True)
    return [content for content, score in recommended if score > 0]  # Return content with scores > 0
# Example: Recommend content for Alice
recommended_for_alice = recommend_content("Alice", user_content_matrix, content_similarity_df)
print("Recommendations for Alice:", recommended_for_alice)
What This Code Does:
Data Input: Simulates user interactions with posts (e.g., likes, views).
User-Content Matrix: Creates a matrix showing how much each user interacted with each post.
Content Similarity: Calculates similarity between posts using cosine similarity.
Recommendations: Combines user interaction data with content similarity to recommend posts Alice hasn’t seen.

Copy
import openai
from flask import Flask, request, jsonify
app = Flask(__name__)
openai.api_key = "YOUR_API_KEY"
@app.route('/generate', methods=['POST'])
def generate_response():
    prompt = request.json.get('prompt')  # Capture the prompt sent by the user
    response = openai.Completion.create(
        engine="text-davinci-003",  # GPT engine
        prompt=prompt,
        max_tokens=150  # Adjust as needed
    )
    return jsonify({"response": response.choices[0].text.strip()})
if __name__ == '__main__':
    app.run(debug=True)
This code listens for POST requests with a user’s prompt and then sends the prompt to GPT, returning the response to the user.

javascript
Copy
import React, { useState } from 'react';
import axios from 'axios';
function App() {
  const [input, setInput] = useState("");
  const [output, setOutput] = useState("");
  const handleSubmit = async () => {
    const response = await axios.post('http://localhost:5000/generate', { prompt: input });
    setOutput(response.data.response);
  };
  return (
    <div>
      <input
        type="text"
        value={input}
        onChange={(e) => setInput(e.target.value)}
        placeholder="Ask your question..."
      />
      <button onClick={handleSubmit}>Submit</button>
      <p>Response: {output}</p>
    </div>
  );
}
export default App;
This code takes user input and sends it to the backend (which communicates with GPT). Once it receives the response, it displays it to the user.
